## 使用

cd SMSBoom-Terminal

sudo apt install python3 python3-venv python3-pip

python3 -m venv env

source env/bin/activate   //Linux
.\env\Scripts\activate   //Windows

pip install requests colorama tqdm urllib3

python SMSBoom.py
